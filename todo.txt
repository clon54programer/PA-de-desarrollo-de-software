--------------------------------------------------------

tablas de la base de datos, soft_pa

user:
    id: PK
    name: VARCHAR(80)
    password: VARCHAR(24)

activosBase:
    id: PK
    name: VARCHAR(80)
    descripcion: VARCHAR(255)
    valor: REAL

activosFisico:
    ubicacion: VARCHAR(255)
    estado: enum(activo,en mantenimiento, retirao)

activosDigital:
    licencia: VARCHAR(255)
    version: VARCHAR(20)

vulnerabilidad:
    ID: PK
    activo_afectado: FK
    descripcion: TEXT
    cve: VARCHAR(20)
    fecha_de_descubrimiento:
    estado: enum(parchada,no_parchada)
    recomendaciones: Lista de FK

recomendacion:
    id: PK
    descripcion: TEXT
